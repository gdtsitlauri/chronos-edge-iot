\begin{abstract}
The proliferation of Internet-of-Things (IoT) devices at the network edge demands intelligent orchestration of computation, communication, and learning under stringent latency, energy, and bandwidth constraints. Existing approaches address federated learning, task offloading, and communication optimization largely in isolation, neglecting the higher-order causal dependencies that govern real-world edge ecosystems. We propose CHRONOS (Causal Hypergraph-Regulated Orchestration of Networked Offloading with Spiking Intelligence), a novel algorithmic framework that unifies these dimensions through three foundational contributions: (1) a Causal Hypergraph State Representation (CHSR) that models the edge-IoT system as a dynamic hypergraph with causal interventional relationships; (2) a Spiking Multi-Agent Reinforcement Learning (S-MARL) architecture for event-driven, energy-efficient decision-making; and (3) a Constrained Causal Multi-Objective Markov Game (CC-MOMG) formulation solved via a novel causal counterfactual policy gradient. CHRONOS is validated on a Digital Twin simulation platform. All research and experiments were conducted in 2026.
\end{abstract}

\section{System Architecture}
\subsection{System Model}
Consider an edge-IoT system comprising:
\begin{itemize}
	\item A set of edge nodes $\mathcal{N} = \{1, 2, \dots, N\}$, each with compute capacity $c_i$, memory $m_i$, and energy budget $E_i^{\max}$.
	\item A set of IoT devices $\mathcal{D} = \{1, 2, \dots, D\}$ generating tasks and local data.
	\item A set of tasks $\mathcal{T}(t) = \{T_1(t), \dots, T_K(t)\}$ arriving dynamically, each with tuple $T_k = (\delta_k, \omega_k, d_k, \tau_k^{\max})$.
	\item A set of wireless channels $\mathcal{C} = \{1, \dots, C\}$ with time-varying channel gains $h_{ij}^c(t)$.
	\item A federated learning process over a global model $\mathbf{w}$ with local datasets $\{D_i\}_{i=1}^N$ (non-IID).
\end{itemize}

\subsection{Causal Hypergraph Representation}
The system state is a dynamic causal hypergraph:
$$\mathcal{H}(t) = (\mathcal{V}(t), \mathcal{E}(t), \mathcal{W}(t), \mathcal{G}(t))$$
where $\mathcal{V}(t)$ is the vertex set (nodes, devices, tasks, channels), $\mathcal{E}(t)$ is the hyperedge set (causal interactions), $\mathcal{W}(t)$ assigns causal weights, and $\mathcal{G}(t)$ is a structural causal model (SCM) over the hypergraph.

\subsection{Decision Variables}
At each epoch $t$, the system determines:
\begin{itemize}
	\item Task offloading $\mathbf{x}(t)$, resource allocation $\mathbf{r}(t)$
	\item Channel assignment/power $\mathbf{p}(t), \mathbf{a}(t)$
	\item Federated learning params $\boldsymbol{\theta}(t)$, spiking encoding $\boldsymbol{\phi}(t)$
\end{itemize}
Composite action: $\mathbf{A}(t) = (\mathbf{x}(t), \mathbf{r}(t), \mathbf{p}(t), \mathbf{a}(t), \boldsymbol{\theta}(t), \boldsymbol{\phi}(t))$

\subsection{Objectives}
\textbf{1. Federated Model Accuracy (maximize):}
$$f_1(\mathbf{A}) = -\left[ F(\mathbf{w}^*) - F(\mathbf{w}^{(R)}) \right], \quad F(\mathbf{w}) = \sum_{i=1}^{N} \frac{|D_i|}{|D|} F_i(\mathbf{w})$$
\textbf{2. End-to-End Latency (minimize):}
$$f_2(\mathbf{A}) = \max_{k \in \mathcal{T}(t)} \left[ \frac{\delta_k}{R_{ij}^c(t)} + \frac{\omega_k}{c_j \cdot r_{kj}(t)} + \tau_k^{\text{queue}}(t) \right]$$
\textbf{3. Energy Consumption (minimize):}
$$f_3(\mathbf{A}) = \sum_{i \in \mathcal{D}} \left[ \sum_c p_i^c(t) \cdot \frac{\delta_k}{R_{ij}^c(t)} + \kappa_i \cdot \gamma_i \cdot f_i^2 \cdot \omega_i^{\text{local}} \right] + \sum_{j \in \mathcal{N}} E_j^{\text{comp}}(t)$$
\textbf{4. Communication Cost (minimize):}
$$f_4(\mathbf{A}) = \sum_{i=1}^{N} z_i(t) \cdot \rho_i(t) \cdot \|\nabla F_i(\mathbf{w}_i^{(t)})\|_0 \cdot b$$

\subsection{Constraints}
\begin{align*}
&\text{C1 (Deadline):} \quad \forall k: \tau_k^{\text{total}}(t) \leq \tau_k^{\max} \\
&\text{C2 (Energy budget):} \quad \forall i: \sum_{t=0}^{T} E_i(t) \leq E_i^{\max} \\
&\text{C3 (Compute capacity):} \quad \forall j: \sum_{k} r_{kj}(t) \leq 1 \\
&\text{C4 (Task assignment):} \quad \forall k: \sum_{j=0}^{N} x_{kj}(t) = 1 \\
&\text{C5 (Power):} \quad \forall i: \sum_c p_i^c(t) \leq P_i^{\max} \\
&\text{C6 (Causality):} \quad \mathbf{A}(t) \in \mathcal{A}^{\text{causal}}(\mathcal{H}(t))
\end{align*}

\subsection{Formalization}
The problem is a Constrained Causal Multi-Objective Markov Game (CC-MOMG):
$$\min_{\pi_1, \dots, \pi_N} \quad \mathbf{F}(\boldsymbol{\pi}) = \left(-f_1, f_2, f_3, f_4\right)$$
subject to C1–C6, where $\pi_i: \mathcal{S}_i \times \mathcal{H}(t) \rightarrow \Delta(\mathcal{A}_i)$.

\section{Methodology}
\subsection{Spiking Multi-Agent RL}
Each edge agent employs a spiking neural network policy for event-driven, energy-efficient decision-making, communicating via spike-coded messages over the hypergraph attention mechanism.

\subsection{Joint Optimization}
The joint optimization is solved via a causal counterfactual policy gradient method with provable convergence under non-IID, time-varying conditions.

\subsection{Experimental Setup}
Experiments are run on a digital twin simulation platform, calibrated to real-world edge deployments. Benchmarks include multi-seed studies, ablation, and significance tests.
